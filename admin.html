    // Toplu konum ekle
    function addBulkLocations() {
      const bulkText = document.getElementById('bulkLocations').value;
      if (!bulkText.trim()) {
        alert('L√ºtfen konum bilgilerini girin!');
        return;
      }
      
      const lines = bulkText.split('\n').filter(line => line.trim());
      let successCount = 0;
      
      lines.forEach((line, index) => {
        // Satƒ±r ba≈üƒ±ndaki numaralarƒ± temizle (1. 2. 3. vb.)
        let cleanedLine = line.replace(/^\d+\.\s*/, '');
        
        // @ karakterinin etrafƒ±ndaki bo≈üluklarƒ± temizle ama konum adƒ±nƒ±n i√ßindeki bo≈üluklara dokunma
        cleanedLine = cleanedLine.replace(/\s*@\s*/g, '@');
        
        // Google Maps linkini tespit et ve i√ßindeki @ karakterlerini ge√ßici olarak deƒüi≈ütir
        cleanedLine = cleanedLine.replace(/(https:\/\/(?:maps\.google|www\.google\.com\/maps)[^@]*?)@([^@]*?)(@|$)/g, '$1__TEMP__$2$3');
        
        const parts = cleanedLine.split('@');
        if (parts.length < 2) {
          return;
        }
        
        const name = parts[0].trim();
        let link = parts[1].trim();
        const photos = [];
        
        // Google Maps linkindeki ge√ßici deƒüi≈ütirmeleri geri al
        link = link.replace(/__TEMP__/g, '@');
        
        // Fotoƒüraflarƒ± topla
        for (let i = 2; i < parts.length && i < 5; i++) {
          if (parts[i].trim() && parts[i].trim().startsWith('https://')) {
            photos.push(parts[i].trim());
          }
        }
        
        if (!name || !link) {
          return;
        }
        
        // Konumu ekle
        const locations = loadLocations();
        const newLocation = {
          name: name,
          photos: photos,
          link: link
        };
        
        locations.push(newLocation);
        saveLocations(locations);
        successCount++;
      });
      
      renderLocations();
      document.getElementById('bulkLocations').value = '';
    }
